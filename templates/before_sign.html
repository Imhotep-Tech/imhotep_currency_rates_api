fetch(url)
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP error! Status: ${response.status}`);
    }
    return response.json();
  })
  .then(data => {
    console.log('Exchange rates:', data);
    // Example: Convert 100 USD to EUR
    const euroRate = data.data.EUR;
    const amountInEuros = 100 * euroRate;
    console.log(`100 USD = ${amountInEuros.toFixed(2)} EUR`);
  })
  .catch(error => {
    console.error('Error fetching exchange rates:', error);
  });

<?php
// API configuration
$apiKey = 'your_api_key';
$baseCurrency = 'USD';
$url = "https://imhotepexchangeratesapi.pythonanywhere.com/latest_rates/{$apiKey}/{$baseCurrency}";

// Make API request
$ch = curl_init($url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_TIMEOUT, 10);
$response = curl_exec($ch);

if (curl_errno($ch)) {
    echo 'Error fetching exchange rates: ' . curl_error($ch);
} else {
    // Process the data
    $data = json_decode($response, true);
    echo "Exchange rates (base: {$data['meta']['base_currency']})\n";
    echo "Last updated: {$data['meta']['last_updated_at']}\n";
    
    // Example: Convert 100 USD to EUR
    $euroRate = $data['data']['EUR'];
    $amountInEuros = 100 * $euroRate;
    echo "100 USD = " . number_format($amountInEuros, 2) . " EUR";
}

curl_close($ch);
?>
